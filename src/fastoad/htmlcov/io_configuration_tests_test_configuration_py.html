218def test_set_optimization_definition(cleanup): 

219 """ 

220 Tests the modification of the optimization definition in the configuration file 

221 """ 

222 for extension in ["toml", "yml"]: 

223 clear_openmdao_registry() 

224 reference_file = pth.join(DATA_FOLDER_PATH, "valid_sellar.%s" % extension) 

225 editable_file = pth.join(RESULTS_FOLDER_PATH, "editable_valid_sellar.%s" % extension) 

226 

227 conf = FASTOADProblemConfigurator(reference_file) 

228 

229 optimization_def = { 

230 "design_variables": { 

231 "x": {"name": "x", "lower": 0, "upper": 20}, 

232 "z": {"name": "z", "lower": 0, "upper": 10}, 

233 }, 

234 "constraints": { 

235 "gg1": {"name": "gg1", "upper": 10}, 

236 "gg2": {"name": "gg2", "upper": 0}, 

237 }, 

238 "objective": {"f": {"name": "f"}}, 

239 } 

240 

241 optimization_conf = { 

242 "design_variables": [ 

243 {"name": "x", "lower": 0, "upper": 20}, 

244 {"name": "z", "lower": 0, "upper": 10}, 

245 ], 

246 "constraints": [{"name": "gg1", "upper": 10}, {"name": "gg2", "upper": 0}], 

247 "objective": [{"name": "f"}], 

248 } 

249 

250 read = tomlkit.loads if extension == "toml" else YAML(typ="safe").load 

251 

252 with open(reference_file, "r") as file: 

253 d = file.read() 

254 conf_dict = read(d) 

255 conf_dict_opt = conf_dict["optimization"] 

256 # Should be different 

257 assert optimization_conf != conf_dict_opt 

258 

259 conf.set_optimization_definition(optimization_def) 

260 conf.save(editable_file) 

261 with open(editable_file, "r") as file: 

262 d = file.read() 

263 conf_dict = read(d) 

264 conf_dict_opt = conf_dict["optimization"] 

265 # Should be equal 

266 assert optimization_conf == conf_dict_opt
