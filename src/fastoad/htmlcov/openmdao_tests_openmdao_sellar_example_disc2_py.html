18class Disc2(om.ExplicitComponent): 

19 """An OpenMDAO component to encapsulate Disc2 discipline""" 

20 

21 def setup(self): 

22 self.add_input( 

23 "z", val=[5, 2], desc="variable z", units="m**2" 

24 ) # for testing non-None units 

25 self.add_input("y1", val=1.0, desc="") 

26 

27 self.add_output("y2", val=1.0, desc="") 

28 

29 def setup_partials(self): 

30 self.declare_partials("*", "*", method="fd") 

31 

32 def compute(self, inputs, outputs, discrete_inputs=None, discrete_outputs=None): 

33 """ 

34 Evaluates the equation 

35 y2 = y1**(.5) + z1 + z2 

36 """ 

37 

38 z1 = inputs["z"][0] 

39 z2 = inputs["z"][1] 

40 y1 = inputs["y1"] 

41 

42 # Note: this may cause some issues. However, y1 is constrained to be 

43 # above 3.16, so lets just let it converge, and the optimizer will 

44 # throw it out 

45 if y1.real < 0.0: 

46 y1 *= -1 

47 

48 outputs["y2"] = y1 ** 0.5 + z1 + z2
