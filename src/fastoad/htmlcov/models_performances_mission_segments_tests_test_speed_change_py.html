24def test_acceleration_to_TAS(polar): 

25 propulsion = FuelEngineSet(DummyEngine(0.5e5, 1.0e-5), 2) 

26 

27 segment = SpeedChangeSegment( 

28 target=FlightPoint(true_airspeed=250.0), 

29 propulsion=propulsion, 

30 reference_area=120.0, 

31 polar=polar, 

32 thrust_rate=1.0, 

33 time_step=0.2, 

34 engine_setting=EngineSetting.CLIMB, # The engine model does not use this setting 

35 ) 

36 flight_points = segment.compute_from( 

37 FlightPoint(altitude=5000.0, true_airspeed=150.0, mass=70000.0) 

38 ) 

39 

40 last_point = flight_points.iloc[-1] 

41 # Note: reference values are obtained by running the process with 0.01s as time step 

42 assert_allclose(last_point.time, 103.3, rtol=1e-2) 

43 assert_allclose(last_point.altitude, 5000.0) 

44 assert_allclose(last_point.true_airspeed, 250.0) 

45 assert_allclose(last_point.mass, 69896.0, rtol=1e-4) 

46 assert_allclose(last_point.ground_distance, 20697.0, rtol=1e-3) 

47 assert last_point.engine_setting == EngineSetting.CLIMB
