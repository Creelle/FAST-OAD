56class Variable(Hashable): 

57 """ 

58 A class for storing data of OpenMDAO variables. 

59 

60 Instantiation is expected to be done through keyword arguments only. 

61 

62 Beside the mandatory parameter 'name, kwargs is expected to have keys 

63 'value', 'units' and 'desc', that are accessible respectively through 

64 properties :meth:`name`, :meth:`value`, :meth:`units` and :meth:`description`. 

65 

66 Other keys are possible. They match the definition of OpenMDAO's method 

67 :meth:`Component.add_output` described 

68 `here <http://openmdao.org/twodocs/versions/latest/_srcdocs/packages/core/ 

69 component.html#openmdao.core.component.Component.add_output>`_. 

70 

71 These keys can be listed with class method :meth:`get_openmdao_keys`. 

72 **Any other key in kwargs will be silently ignored.** 

73 

74 Special behaviour: :meth:`description` will return the content of kwargs['desc'] 

75 unless these 2 conditions are met: 

76 - kwargs['desc'] is None or 'desc' key is missing 

77 - a description exists in FAST-OAD internal data for the variable name 

78 Then, the internal description will be returned by :meth:`description` 

79 

80 :param kwargs: the attributes of the variable, as keyword arguments 

81 """ 

82

87 _loaded_descriptions = set()

92 def __init__(self, name, **kwargs): 

93 super().__init__() 

94 

95 self.name = name 

96 """ Name of the variable """ 

97 

98 self.metadata: Dict = {} 

99 """ Dictionary for metadata of the variable """ 

100 

101 # Initialize class attributes once at first instantiation ------------- 

102 if not self._base_metadata: 

103 # Get variable base metadata from an ExplicitComponent 

104 comp = om.ExplicitComponent() 

105 # get attributes 

106 metadata = comp.add_output(name="a") 

107 

108 self.__class__._base_metadata = metadata 

109 self.__class__._base_metadata["val"] = 1.0 

110 self.__class__._base_metadata["tags"] = set() 

111 self.__class__._base_metadata["shape"] = None 

112 # Done with class attributes ------------------------------------------ 

113 

114 # Feed self.metadata with kwargs, but remove first attributes with "Unavailable" as 

115 # value, which is a value that can be provided by OpenMDAO. 

116 self.metadata = self.__class__._base_metadata.copy() 

117 self.metadata.update( 

118 { 

119 key: value 

120 for key, value in kwargs.items() 

121 # The isinstance check is needed if value is a numpy array. In this case, a 

122 # FutureWarning is issued because it is compared to a scalar. 

123 if not isinstance(value, str) or value != "Unavailable" 

124 } 

125 ) 

126 

127 if "value" in self.metadata: 

128 self.metadata["val"] = self.metadata.pop("value") 

129 if "description" in self.metadata: 

130 self.metadata["desc"] = self.metadata.pop("description") 

131 

132 self._set_default_shape() 

133 

134 # If no description, use the one from self._variable_descriptions, if available 

135 if not self.description and self.name in self._variable_descriptions: 

136 self.description = self._variable_descriptions[self.name]

138 @classmethod 

139 def read_variable_descriptions(cls, file_parent: str, update_existing: bool = True): 

140 """ 

141 Reads variable descriptions in indicated folder or package, if it contains some. 

142 

143 The file variable_descriptions.txt is looked for. Nothing is done if it is not 

144 found (no error raised also). 

145 

146 Each line of the file should be formatted like:: 

147 

148 my:variable||The description of my:variable, as long as needed, but on one line. 

149 

150 :param file_parent: the folder path or the package name that should contain the file 

151 :param update_existing: if True, previous descriptions will be updated. 

152 if False, previous descriptions will be erased. 

153 """ 

154 if not update_existing: 

155 cls._variable_descriptions = {} 

156 cls._loaded_descriptions = set() 

157 

158 variable_descriptions = None 

159 description_file = None 

160 

161 if file_parent: 

162 if pth.isdir(file_parent): 

163 file_path = pth.join(file_parent, DESCRIPTION_FILENAME) 

164 if pth.isfile(file_path): 

165 description_file = open(file_path) 

166 else: 

167 # Then it is a module name 

168 if DESCRIPTION_FILENAME in PackageReader(file_parent).contents: 

169 description_file = open_text(file_parent, DESCRIPTION_FILENAME) 

170 

171 if description_file is not None: 

172 try: 

173 variable_descriptions = np.genfromtxt( 

174 description_file, delimiter="||", dtype=str, autostrip=True 

175 ) 

176 except Exception as exc: 

177 # Reading the file is not mandatory, so let's just log the error. 

178 _LOGGER.error( 

179 "Could not read file %s in %s. Error log is:\n%s", 

180 DESCRIPTION_FILENAME, 

181 file_parent, 

182 exc, 

183 ) 

184 description_file.close() 

185 

186 if variable_descriptions is not None: 

187 if np.shape(variable_descriptions) == (2,): 

188 # If the file contains only one line, np.genfromtxt() will return a (2,)-shaped 

189 # array. We need a reshape for dict.update() to work correctly. 

190 variable_descriptions = np.reshape(variable_descriptions, (1, 2)) 

191 

192 cls._loaded_descriptions.add(file_parent) 

193 cls.update_variable_descriptions(variable_descriptions) 

194 _LOGGER.info("Loaded variable descriptions in %s", file_parent) 

195

280 def _set_default_shape(self): 

281 """Automatically sets shape if not set""" 

282 if self.metadata["shape"] is None: 

283 shape = np.shape(self.value) 

284 if not shape: 

285 shape = (1,)

288 def __eq__(self, other): 

289 # same arrays with nan are declared non equals, so we need a workaround 

290 my_metadata = dict(self.metadata) 

291 other_metadata = dict(other.metadata) 

292 my_value = np.asarray(my_metadata.pop("val")) 

293 other_value = np.asarray(other_metadata.pop("val")) 

294 

295 # Let's also ignore unimportant keys 

296 for key in METADATA_TO_IGNORE: 

297 if key in my_metadata: 

298 del my_metadata[key] 

299 if key in other_metadata: 

300 del other_metadata[key] 

301 

302 return ( 

303 isinstance(other, Variable) 

304 and self.name == other.name 

305 and np.all(np.isclose(my_value, other_value, equal_nan=True)) 

306 and my_metadata == other_metadata 

307 )
