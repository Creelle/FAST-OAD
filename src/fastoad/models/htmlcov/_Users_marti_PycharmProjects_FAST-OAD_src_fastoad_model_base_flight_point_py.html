24@dataclass 

25class FlightPoint: 

26 """ 

27 Dataclass for storing data for one flight point. 

28 

29 This class is meant for: 

30 

31 - pandas friendliness: data exchange with pandas DataFrames is simple 

32 - extensibility: any user might add fields to the **class** using :meth:`add_field` 

33 

34 **Exchanges with pandas DataFrame** 

35 

36 A pandas DataFrame can be generated from a list of FlightPoint instances:: 

37 

38 >>> import pandas as pd 

39 >>> from fastoad.model_base import FlightPoint 

40 

41 >>> fp1 = FlightPoint(mass=70000., altitude=0.) 

42 >>> fp2 = FlightPoint(mass=60000., altitude=10000.) 

43 >>> df = pd.DataFrame([fp1, fp2]) 

44 

45 And FlightPoint instances can be created from DataFrame rows:: 

46 

47 # Get one FlightPoint instance from a DataFrame row 

48 >>> fp1bis = FlightPoint.create(df.iloc[0]) 

49 

50 # Get a list of FlightPoint instances from the whole DataFrame 

51 >>> flight_points = FlightPoint.create_list(df) 

52 

53 **Extensibility** 

54 

55 FlightPoint class is bundled with several fields that are commonly used in trajectory 

56 assessment, but one might need additional fields. 

57 

58 Python allows to add attributes to any instance at runtime, but for FlightPoint to run 

59 smoothly, especially when exchanging data with pandas, you have to work at class level. 

60 This can be done using :meth:`add_field`, preferably outside of any class or function:: 

61 

62 # Adds a float field with None as default value 

63 >>> FlightPoint.add_field("ion_drive_power") 

64 

65 # Adds a field and define its type and default value 

66 >>> FlightPoint.add_field("warp", annotation_type=int, default_value=9) 

67 

68 # Now these fields can be used at instantiation 

69 >>> fp = FlightPoint(ion_drive_power=110.0, warp=12) 

70 

71 # Removes a field, even an original one (useful only to avoid having it in outputs) 

72 >>> FlightPoint.remove_field("sfc") 

73 

74 .. note:: 

75 

76 All parameters in FlightPoint instances are expected to be in SI units. 

77 

78 """

154 @classmethod 

155 def add_field(cls, name: str, annotation_type=float, default_value: Any = None, unit=None): 

156 """ 

157 Adds the named field to FlightPoint class. 

158 

159 If the field name already exists, the field is redefined. 

160 

161 :param name: field name 

162 :param annotation_type: field type 

163 :param default_value: field default value 

164 :param unit: expected unit for the added field ("-" should be provided for a dimensionless 

165 physical quantity) 

166 """ 

167 cls.remove_field(name) 

168 

169 del cls.__init__ # Delete constructor to allow it being rebuilt with dataclass() call 

170 setattr(cls, name, default_value) 

171 cls.__annotations__[name] = annotation_type 

172 dataclass(cls) 

173 if unit: 

174 cls._units[name] = unit

176 @classmethod 

177 def remove_field(cls, name): 

178 """ 

179 Removes the named field from FlightPoint class. 

180 

181 :param name: field name 

182 """ 

183 if name in cls.__annotations__: 

184 del cls.__init__ # Delete constructor to allow it being rebuilt with dataclass() call 

185 delattr(cls, name) 

186 del cls.__annotations__[name] 

187 dataclass(cls) 

188 if name in cls._units: 

189 del cls._units[name]

192 """
